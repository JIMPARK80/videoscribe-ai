    runtime_hooks=[],
    excludes=[],

    optimize=0,

    pyz,
    a.scripts,
    [],
    exclude_binaries=True,
    name='VideoToTextConverter',
    debug=False,
    bootloader_ignore_signals=False,
    strip=False,
    upx=False,
    upx_exclude=[],
    runtime_tmpdir=None,
